## Install

```bash
npm install --save use-activity-dispatcher
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'use-activity-dispatcher'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
